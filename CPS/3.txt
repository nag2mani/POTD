Given a string, remove all the vowels from the string.
E.g, "hello world" => "hll wrld"

def removeVowels(s):
    new_str = ''
    vowels = ['a', 'e', 'i', 'o','u']
    for c in s:
        if c not in vowels:
            new_str += c
