# Given an array as input, print out the array backwards.
# Ignore empty and null arrays.

//Nagmani
def printArrayBack(arr):
    new_arr = []
    for i in range(len(arr)-1, -1, -1):
        new_arr.append(arr[i])
    return new_arr

print(printArrayBack(arr))


//Nagmani
def printBackward(arr):
    for i in range(len(arr)-1, -1, -1):
        print(arr[i])
printBackward(arr)
