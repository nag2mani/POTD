Q. Given two string check if it is it's permutation.

def checkPermutation(s1, s2):
    if len(s1) != len(s2):
        return False
    frq_s1 = Counter(s1)
    frq_s2 = Counter(s2)
    for key in frq_s1:
        if key not in frq_s2:
            return False
        else:
            if frq_s1[key] != frq_s2[key]:
                return False
    return True
